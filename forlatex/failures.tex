\begin{itemize}
	\item Given $A,B\in\mathcal{S}^n$ (symmetric $n\times n$ matrices),
		$\tr(A)\geq 0$ and $\tr(B)<0$, then there exists $w\in\mathbb{R}^n$, such that
		$w^TAw\geq 0$ and $w^TBw<0$. (Optimization, CUHK, 2013, sem 1)
\end{itemize}
